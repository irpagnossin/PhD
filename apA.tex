\hspace{-\parindent}\rule{2.5cm}{0cm}
\begin{list}{}
	{
	\setlength\leftmargin{2.5cm}
	\setlength\labelwidth{2.5cm}
	\setlength\labelsep{0.5cm}
	\setlength\parsep{0.5ex plus 0.2ex minus 0.1ex}
	\setlength\itemsep{0cm}
	\setlength\itemindent{0cm}
	}


	\item[$\vetor A$]
		Potencial vetor associado ao campo magnético: $\vetor B=\vetor\nabla\times\vetor A$;

	\item[$A_{ee}$]
		Coeficiente angular do gráfico $\tauphi^{-1} \times T$;

	\item[$B=|\vetor B|$]
		Campo magnético aplicado;

	\item[$b_{\varphi,\text{\tiny SO}}$]
		Parâmetro auxiliar de ajuste, definido como $B_{\varphi,\text{\tiny SO}}/B$;

	\item[$B_{\varphi,\text{\tiny SO}}$]
		Campo magnético no qual um elétron percorre, por difusão e durante o tempo $\tauphi$ ($\tauSO$), uma trajetória que encerra uma área tal que contenha aproximadamente um \foreign{quantum} de fluxo magnético ($\Phi_0$);


	\item[$d_\text{InAs}$]
		Espessura nominal da camada de InAs, em monocamadas (MC);

	\item[$\delta(x)$]
		Função delta de Dirac;

	\item[$\delta$-Si]
		Camada de dopagem planar

	\item[$\delta\sigma$]
		Correção \emph{teórica} da condutividade clássica, devido à localização-fraca ou interações entre elétrons;

	\item[$\Delta\sigma$]
		Correção \emph{experimental} da condutividade clássica, devido à localização-fraca ou interações entre elétrons;

	\item[$\Delta(1/B)$]
		Período das oscilações de magnetorresistência (efeito Shubnikov-de Haas);

	\item[$\Delta_z$]
		\foreign{Zeeman splitting} (\foreign{gap} de \foreign{spin});

	\item[$e$]
		Carga do elétron (\unit{1,602\times\power{10}{-19}}{\coulomb});

	\item[$E=|\vetor E|$]
		Campo elétrico;

	\item[$E_F$ ou $\zeta$ ($\zeta^0$)]
		Nível de Fermi (em campo magnético nulo);

	\item[$E_x$]
		Campo elétrico paralelo a $I$ (dá origem a $I$);

	\item[$E_y$ ou $E_H$]
		Campo elétrico perpendicular a $I$ (efeito Hall);

	\item[$\epsilon\doteq\epsilon(\vetor k)$]	
		Energia do elétron no orbital (estado) $\vetor k$;

	\item[$\epsilon_s$]
		Energia do fundo da banda $s$;

	\item[$F_0^\sigma$]
		Constante do líquido de Fermi associado ao estado triplete;

	\item[$F_t$ e $F_s$]
		Contribuições dos estados triplete e singlete do elétron, respectivamente, para a correção da condutividade clássica (devido à localização-fraca);

	\item[$\formFactor \doteq l_x/l_y$]
		Fator de forma (relaciona a resistência com a resistividade);

	\item[$G$]
		Condutância;

	\item[$\Gamma$]
		Ponto $\vetor k = \vetor 0$ da primeira zona de Brillouin;

	\item[$\Gamma_\ell$]
		Largura natural do nível de Landau $\ell$;

	\item[$h=2\pi\hbar$]
		Constante de Planck (\unit{6,626\times\power{10}{-34}}{\joule\cdot\second});


	\item[$I$]
		Corrente elétrica que percorre a amostra;

	\item[$\vetor j$]
		Densidade de corrente, $I=\int\limits_{\Sigma} \vetor j\cdot d\vetor\Sigma$;		

	\item[$\vetor k$]		
		Estado, orbital ou vetor de onda de um elétron;
		
	\item[$k_B$]
		Constante de Boltzmann (\unit{1,381\times\power{10}{-23}}{\joule\per\kelvin});
	
	\item[$\vetor k_F$]
		Vetor de onda de Fermi;
	
	\item[$k_x,k_y,k_z$]
		Componentes cartesianos de $\vetor k$;

	\item[$l_x$, $l_y$, $l_z$]
		Dimensões da região de interesse nas direções $x$, $y$ e $z$, respectivamente;

	\item[$\ell$]
		Número quântico orbital do nível de Landau ou, equivalentemente, índice de identificação do $\ell$-ésimo nível de Landau;

	\item[$L$]
		Comprimento do canal condutor (poço-quântico);

	\item[$\lambda$]
		Constante de interação;

	\item[$m_0$]
		Massa de repouso do elétron (\unit{9,109\times\power{10}{-31}}{\kilo\gram});

	\item[$\meff_s$]
		Massa efetiva do elétron;

	\item[$M$]
		Quantidade de estados de borda contra-rotativos;

	\item[$N$]
		Quantidade de níveis de Landau totalmente preenchidos (abaixo do nível de Landau) ou a quantidade de estados de borda normais;

	\item[$n$ ou $N_{2D}$]
		Concentração de portadores no canal bidimensional (poço-quântico);

	\item[$n_\delta$]
		Densidade de portadores no plano da dopagem;

	\item[$N_{0D}$]
		Densidade de elétrons armadilhados nos pontos-quânticos;

	\item[$\mu \doteq e\tau/\mEff$]
		Mobilidade de transporte (obtida do efeito Hall clássico);

	\item[$\mu_q \doteq e\tau_q/\mEff$]
		Mobilidade quântica (obtida do efeito Shubnikov-de Haas);

	\item[$\mu_\delta$]
		Mobilidade de transporte dos portadores no plano da dopagem;

	\item[$\nu$]
		Fator de preenchimento dos níveis de Landau ou freqüência;

	\item[$p$]
		Expoente da dependência de $\tauphi$ com $T$: $\tauphi \propto T^{-p}$;


	\item[$\vetor p$]
		Momento linear do elétron (classicamente, igual a $m\vetor v$; quanticamente, $\hbar \dot{\vetor k}$);


	\item[$\Phi_0 \doteq h/e$]
		\foreign{Quantum} de fluxo magnético;

	\item[$R_0$]
		Resistência em campo nulo;


	\item[$R_H \doteq V_H/I$]
		Resistência Hall;

	\item[$R_K \doteq h/e^2$]
		Constante de Von-Klitzing ($\unit{25 812,807}{\ohm}$);

	\item[$\rho$]
		Resistividade;

	\item[$\rho_0$]
		Resistividade em campo magnético nulo ($=R_0/\formFactor$);

	\item[$r_s$]
		Razão entre as energias de interação coulombiana e cinética do elétron;

	\item[$\sigma$]
		Condutividade;

	\item[$\sigma_0 \doteq 1/\rho_0$]
		Condutividade em campo magnético nulo (a definição acima não vale quando $B \ne 0$);

	\item[$T$]
		Temperatura, probabilidade de transmissão ou matriz de transmissão;

	\item[$T_{ij}$]
		Quantidade de modos transversais que atingem o contato $i$, partindo de $j$;

	\item[$\tau$]
		Tempo de relaxação do momento ou, equivalentemente, tempo médio de vida num autoestado do momento;

	\item[$\tau_{ee}$]
		Tempo de espalhamento devido a interações entre elétrons;

	\item[$\tau_q$]
		Tempo quântico (\foreign{single-particle ralaxation time}) ou, equivalentemente, tempo de vida do \foreign{quasi}-elétron num nível de Landau;

	\item[$\tau_\text{\tiny QD}$]
		Tempo de espalhamento dos elétrons do 2DES por aqueles armadilhados pelos QDs;

	\item[$\tauSO$]
		Tempo de espalhamento \foreign{spin}-órbita;

	\item[$\tauphi$]
		Tempo de coerência da função de onda do elétron;

	\item[$\tauphi^{-1}$]
		Taxa de decoerência ou, equivalentemente, de quebra de fase da função de onda do elétron;



	\item[$\vetor v$]
		Velocidade de arrasto;

	\item[$V_{GS}$]
		Tensão entre a fonte e a porta (\foreign{gate}-\foreign{source});

	\item[$V_{DS}$]
		Tensão entre a fonte e o dreno (\foreign{drain}-\foreign{source});

	\item[$V_H$]
		Tensão Hall (medida perpendicularmente a $I$);



	\item[$\varg$ e $\varg^*(B)$]
		Fator de Landè de um elétron e efetivo (amplificado pelas interações de troca);

	\item[$\varsigma = \pm 1/2$]
		Número quântico de \foreign{spin};

	\item[$\varrho$]
		Densidade de estados;

	\item[$\omega_c=eB/\meff$]
		Freqüência ciclotrônica;		

	\item[$\psi_s$ e $\psi_\delta$]
		Função de onda do elétron na sub-banda $s$ e no plano da dopagem, respectivamente;

\end{list}
